**Goal:** Spostare la pagina Radio su `/radio`, aggiungere una pagina dettaglio episodio con tracklist completa da 1001Tracklists (dinamica via serverless), rimuovere i filtri e aggiungere il logo rotante accanto all’H1.

**Tech Stack:** React + TypeScript + React Router, Vite, Vercel Functions (Node), Tailwind CSS.

export type RadioEpisode = {
  episodeCode: string | null;
  title: string;
  publishedAt: string | null;
  durationSec: number | null;
  soundcloudUrl: string | null;
  audioUrl: string | null;
  coverUrl: string | null;
  summary: string | null;
  tracklistUrl: string | null;
};

export function extractEpisodeCode(input: string): string | null {
  const m = input.match(/\bEP\s?(\d{1,4})\b/i);
  if (!m) return null;
  return `ep${String(Number(m[1]))}`;
}

export function extractTracklistUrl(text: string): string | null {
  if (!text) return null;
  const m =
    text.match(/https?:\/\/(?:www\.)?1001\.tl\/[^\s"'<>]+/i) ??
    text.match(/https?:\/\/(?:www\.)?1001tracklists\.com\/[^\s"'<>]+/i);
  return m ? m[0] : null;
}

- [ ] **Step 3: Definire i tipi e parser contract (Tracklist domain)**

Creare `src/lib/tracklistParse.ts`:

```ts
export type TracklistItem = {
  index: number;
  artist: string;
  title: string;
  label: string | null;
  timeText: string | null;
  timeSec: number | null;
};

export type ParsedTracklist = {
  sourceUrl: string;
  items: TracklistItem[];
};

function parseTimeToSec(t: string): number | null {
  const m = t.match(/^(\d{1,2}):(\d{2})$/);
  if (!m) return null;
  return Number(m[1]) * 60 + Number(m[2]);
}

export function parse1001TracklistHtml(html: string, sourceUrl: string): ParsedTracklist {
  const items: TracklistItem[] = [];

  const rowRegex =
    /<[^>]*class="[^"]*\btlToogle\b[^"]*"[^>]*>[\s\S]*?<meta[^>]*itemprop="position"[^>]*content="(\d+)"[\s\S]*?<\/[^>]+>[\s\S]*?<span[^>]*class="[^"]*\btrackValue\b[^"]*"[^>]*>([\s\S]*?)<\/span>[\s\S]*?(?:<span[^>]*class="[^"]*\btracklabel\b[^"]*"[^>]*>([\s\S]*?)<\/span>)?[\s\S]*?(?:<span[^>]*class="[^"]*\btracktime\b[^"]*"[^>]*>([\s\S]*?)<\/span>)?/gi;

  let m: RegExpExecArray | null;
  while ((m = rowRegex.exec(html))) {
    const index = Number(m[1]);
    const value = stripTags(m[2]).trim();
    const label = m[3] ? stripTags(m[3]).trim() : null;
    const timeText = m[4] ? stripTags(m[4]).trim() : null;

    const split = value.split(' - ');
    const artist = (split[0] ?? '').trim();
    const title = (split.slice(1).join(' - ') ?? '').trim();

    if (!artist || !title || !Number.isFinite(index)) continue;

    items.push({
      index,
      artist,
      title,
      label: label || null,
      timeText: timeText || null,
      timeSec: timeText ? parseTimeToSec(timeText) : null,
    });
  }

  return { sourceUrl, items };
}

function stripTags(s: string): string {
  return s.replace(/<[^>]+>/g, ' ').replace(/\s+/g, ' ');
}
```

Creare `api/radio-feed.js`:

```js
const FEED_URL = 'https://feeds.soundcloud.com/users/soundcloud:users:1042711684/sounds.rss';

function extractTag(text, tag) {
  const m = text.match(new RegExp(`<${tag}[^>]*>([\\s\\S]*?)<\\/${tag}>`, 'i'));
  return m ? m[1].trim() : null;
}

function decodeCdata(s) {
  return s ? s.replace(/^<!\\[CDATA\\[|\\]\\]>$/g, '') : s;
}

function findAllItems(xml) {
  const items = [];
  const re = /<item>([\s\S]*?)<\/item>/gi;
  let m;
  while ((m = re.exec(xml))) items.push(m[1]);
  return items;
}

function extractEpisodeCode(title) {
  const m = title.match(/\bEP\s?(\d{1,4})\b/i);
  if (!m) return null;
  return `ep${String(Number(m[1]))}`;
}

function extractTracklistUrl(text) {
  if (!text) return null;
  const m =
    text.match(/https?:\/\/(?:www\.)?1001\.tl\/[^\s"'<>]+/i) ||
    text.match(/https?:\/\/(?:www\.)?1001tracklists\.com\/[^\s"'<>]+/i);
  return m ? m[0] : null;
}

export default async function handler(req, res) {
  if (req.method !== 'GET') return res.status(405).json({ error: 'Method not allowed' });

  const ac = new AbortController();
  const t = setTimeout(() => ac.abort(), 10000);

  try {
    const r = await fetch(FEED_URL, { signal: ac.signal });
    if (!r.ok) return res.status(502).json({ error: 'Upstream error' });
    const xml = await r.text();

    const episodes = findAllItems(xml).map((raw) => {
      const title = decodeCdata(extractTag(raw, 'title')) || 'Untitled';
      const pubDate = extractTag(raw, 'pubDate');
      const link = extractTag(raw, 'link');
      const summary = decodeCdata(extractTag(raw, 'description') || extractTag(raw, 'itunes:summary'));
      const duration = extractTag(raw, 'itunes:duration');
      const enclosureUrl = (() => {
        const m = raw.match(/<enclosure[^>]*url="([^"]+)"/i);
        return m ? m[1] : null;
      })();
      const coverUrl = (() => {
        const m = raw.match(/<itunes:image[^>]*href="([^"]+)"/i);
        return m ? m[1] : null;
      })();

      const durationSec = (() => {
        if (!duration) return null;
        const parts = duration.split(':').map((x) => Number(x));
        if (parts.some((n) => Number.isNaN(n))) return null;
        if (parts.length === 3) return parts[0] * 3600 + parts[1] * 60 + parts[2];
        if (parts.length === 2) return parts[0] * 60 + parts[1];
        return null;
      })();

      return {
        episodeCode: extractEpisodeCode(title),
        title,
        publishedAt: pubDate || null,
        durationSec,
        soundcloudUrl: link || null,
        audioUrl: enclosureUrl,
        coverUrl,
        summary: summary || null,
        tracklistUrl: extractTracklistUrl(summary || ''),
      };
    });

    res.setHeader('Cache-Control', 'public, s-maxage=600, stale-while-revalidate=3600');
    return res.status(200).json({ episodes });
  } catch (e) {
    return res.status(500).json({ error: 'Failed to fetch feed' });
  } finally {
    clearTimeout(t);
  }
}
```

```js
import { parse1001TracklistHtml } from '../src/lib/tracklistParse.ts';

const ALLOW = new Set(['1001.tl', 'www.1001.tl', '1001tracklists.com', 'www.1001tracklists.com']);

function safeUrl(u) {
  try {
    const url = new URL(u);
    if (!ALLOW.has(url.hostname)) return null;
    if (url.protocol !== 'https:' && url.protocol !== 'http:') return null;
    return url;
  } catch {
    return null;
  }
}

export default async function handler(req, res) {
  if (req.method !== 'GET') return res.status(405).json({ error: 'Method not allowed' });

  const raw = req.query?.url;
  const url = typeof raw === 'string' ? safeUrl(raw) : null;
  if (!url) return res.status(400).json({ error: 'Invalid url' });

  const ac = new AbortController();
  const t = setTimeout(() => ac.abort(), 10000);

  try {
    const r = await fetch(url.toString(), {
      signal: ac.signal,
      redirect: 'follow',
      headers: {
        'User-Agent': 'TechMyHouseBot/1.0 (+https://techmyhouse.it)',
        Accept: 'text/html,*/*',
      },
    });

    if (!r.ok) return res.status(502).json({ error: 'Upstream error', items: [] });

    const finalUrl = r.url || url.toString();
    const ct = r.headers.get('content-type') || '';
    if (!ct.includes('text/html')) return res.status(502).json({ error: 'Unexpected content type', items: [] });

    const html = await r.text();
    const parsed = parse1001TracklistHtml(html, finalUrl);

    res.setHeader('Cache-Control', 'public, s-maxage=86400, stale-while-revalidate=604800');
    return res.status(200).json(parsed);
  } catch (e) {
    return res.status(200).json({ sourceUrl: url.toString(), items: [] });
  } finally {
    clearTimeout(t);
  }
}
```
